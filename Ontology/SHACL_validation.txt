@prefix sh:  <http://www.w3.org/ns/shacl#> .
@prefix xsd: <http://www.w3.org/2001/XMLSchema#> .
@prefix :    <http://example.org/menu#> .

# ——————————————————————————————
# Shape per Ingredient
#   • deve avere category (stringa non vuota)
#   • deve avere calories (integer ≥ 0)
#   • può avere zero o più hasAllergen
:IngredientShape
    a sh:NodeShape ;
    sh:targetClass :Ingredient ;

    # category come literal stringa
    sh:property [
        sh:path     :category ;
        sh:datatype xsd:string ;
        sh:minCount 1 ;
        sh:maxLength 30 ;
    ] ;

    # calories come integer ≥0
    sh:property [
        sh:path        :calories ;
        sh:datatype    xsd:integer ;
        sh:minCount    1 ;
        sh:minInclusive 0 ;
    ] ;

    # allergeni (object prop)
    sh:property [
        sh:path     :hasAllergen ;
        sh:class    :Allergen ;
    ] .

# ——————————————————————————————
# Shape per Meal
#   • deve avere almeno un hasIngredient
#   • deve avere totalCalories (integer ≥ 0)
:MealShape
    a sh:NodeShape ;
    sh:targetClass :Meal ;

    # almeno un ingrediente
    sh:property [
        sh:path     :hasIngredient ;
        sh:class    :Ingredient ;
        sh:minCount 1 ;
    ] ;

    # totalCalories come integer ≥0
    sh:property [
        sh:path        :totalCalories ;
        sh:datatype    xsd:integer ;
        sh:minCount    1 ;
        sh:minInclusive 0 ;
    ] .
