# 1. What ingredients does each dish have? (step-by-step navigation)

PREFIX : <http://example.org/menu#>

SELECT ?meal ?ing WHERE {
  ?meal a :Meal .
  ?meal :hasIngredient ?ing .
}

---

# 2. Which dishes contain at least one “dairy” ingredient?

PREFIX : <http://example.org/menu#>

SELECT DISTINCT ?meal WHERE {
  ?meal :hasIngredient ?ing .
  ?ingredient :category ?cat .
  FILTER (?cat = "dairy")
}

---

# 3. Dish → ingredient → allergen (path 2-step)

PREFIX : <http://example.org/menu#>

SELECT DISTINCT ?meal ?all WHERE {
  ?meal :hasIngredient ?ing .
  ?ing :hasAllergen ?all .
}

---

# 4. “Light” dish: less than 500 calories (filter on data-property)

PREFIX : <http://example.org/menu#>

SELECT ?meal ?cals WHERE {
  ?meal :totalCalories ?cals .
  FILTER (?cals <= 500)
}

---

# 5. Extract subgraph of “hasIngredient” (CONSTRUCT)

PREFIX : <http://example.org/menu#>

CONSTRUCT {
  ?meal :hasIngredient ?ing .
}
WHERE {
  ?meal :hasIngredient ?ing .
}

---

# 6. Retrieve any additional properties of an ingredient (OPTIONAL)

PREFIX : <http://example.org/menu#>

SELECT ?ing ?allergen ?cals WHERE {
  ?ing a :Ingredient .
  OPTIONAL { ?ing :hasAllergen ?allergen }
  OPTIONAL { ?ing :calories ?cals }
}
